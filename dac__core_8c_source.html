213  dds_st[phy->id_no].num_buf_channels = 2;

237  }
238  dac_datasel(phy, -1, DATA_SEL_DDS);
239  break;
240  case DATA_SEL_DMA:
241  if(config_dma)
242  {
243  tx_count = sizeof(sine_lut) / sizeof(uint16_t);
244  if(dds_st[phy->id_no].rx2tx2)
245  {
246 #ifdef FMCOMMS5
247  for(index = 0, index_mem = 0; index < (tx_count * 2); index += 2, index_mem += 4)

259

263  index_i2 -= (tx_count * 2);
264  if(index_q2 >= (tx_count * 2))
265  index_q2 -= (tx_count * 2);

267  data_q2 = (sine_lut[index_q2 / 2] << 4);
268  Xil_Out32(DAC_DDR_BASEADDR + (index_mem + 1) * 4, data_i2 | data_q2);
269 #ifdef FMCOMMS5
270  Xil_Out32(DAC_DDR_BASEADDR + (index_mem + 2) * 4, data_i1 | data_q1);
271  Xil_Out32(DAC_DDR_BASEADDR + (index_mem + 3) * 4, data_i2 | data_q2);
272 #endif
273  }
274  }
275  else
276  {
277  for(index = 0; index < tx_count; index += 1)
278  {
279  index_i1 = index;
280  index_q1 = index + (tx_count / 4);
281  if(index_q1 >= tx_count)

284  data_q1 = (sine_lut[index_q1] << 4);

286  }
287  }
288  Xil_DCacheFlush();
289  if(dds_st[phy->id_no].rx2tx2)
290  {
291  length = (tx_count * 8);
292  }
293  else
294  {
295  length = (tx_count * 4);
296  }
297 #ifdef FMCOMMS5
298  length = (tx_count * 16);
299 #endif
300  dac_dma_write(AXI_DMAC_REG_CTRL, 0);
301  dac_dma_write(AXI_DMAC_REG_CTRL, AXI_DMAC_CTRL_ENABLE);
302  dac_dma_write(AXI_DMAC_REG_SRC_ADDRESS, DAC_DDR_BASEADDR);
303  dac_dma_write(AXI_DMAC_REG_SRC_STRIDE, 0x0);

306  dac_dma_write(AXI_DMAC_REG_START_TRANSFER, 0x1);
307  }
308  dac_datasel(phy, -1, DATA_SEL_DMA);
309  break;
310  default:
311  break;
312  }
313  dds_st[phy->id_no].enable = true;
314  dac_start_sync(phy, 0);
315 }

374 void dds_set_scale(struct ad9361_rf_phy *phy, uint32_t chan, int32_t scale_micro_units)
375 {
376  uint32_t scale_reg;
377  uint32_t sign_part;
378  uint32_t int_part;
379  uint32_t fract_part;
380 
381  if (PCORE_VERSION_MAJOR(dds_st[phy->id_no].pcore_version) > 6)
382  {
383  if(scale_micro_units >= 1000000)
384  {
385  sign_part = 0;
386  int_part = 1;
387  fract_part = 0;
388  dds_st[phy->id_no].cached_scale[chan] = 1000000;
389  goto set_scale_reg;
390  }
391  if(scale_micro_units <= -1000000)
392  {
393  sign_part = 1;
394  int_part = 1;
395  fract_part = 0;
396  dds_st[phy->id_no].cached_scale[chan] = -1000000;
397  goto set_scale_reg;
398  }
399  dds_st[phy->id_no].cached_scale[chan] = scale_micro_units;
400  if(scale_micro_units < 0)
401  {
402  sign_part = 1;
403  int_part = 0;
404  scale_micro_units *= -1;
405  }
406  else
407  {
408  sign_part = 0;
409  int_part = 0;
410  }
411  fract_part = (uint32_t)(((uint64_t)scale_micro_units * 0x4000) / 1000000);
412  set_scale_reg:
413  scale_reg = (sign_part << 15) | (int_part << 14) | fract_part;
414  }
415  else
416  {
417  if(scale_micro_units >= 1000000)
418  {
419  scale_reg = 0;
420  scale_micro_units = 1000000;
421  }
422  if(scale_micro_units <= 0)
423  {
424  scale_reg = 0;
425  scale_micro_units = 0;
426  }
427  dds_st[phy->id_no].cached_scale[chan] = scale_micro_units;
428  fract_part = (uint32_t)(scale_micro_units);
429  scale_reg = 500000 / fract_part;
430  }
431  dac_stop(phy);
432  dac_write(phy, DAC_REG_CHAN_CNTRL_1_IIOCHAN(chan), DAC_DDS_SCALE(scale_reg));
433  dac_start_sync(phy, 0);
434 }
